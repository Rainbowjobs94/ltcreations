As Code: Context-Aware Generative Synthesis

import random

def create_like_gemini_advanced(prompt_concepts, context_vector, creativity=0.5):
    weighted_pool = prompt_concepts['nouns'] + context_vector

    if random.random() < creativity:
        chosen_subject = random.choice(weighted_pool)
        chosen_action = random.choice(prompt_concepts['verbs'])
        chosen_object = random.choice(weighted_pool)
    else:
        chosen_subject = weighted_pool[0]
        chosen_action = prompt_concepts['verbs'][0]
        chosen_object = weighted_pool[1] if len(weighted_pool) > 1 else weighted_pool[0]

    return f"The {chosen_subject} {chosen_action} the {chosen_object}."
